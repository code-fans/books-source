只需穷举三个数即可，即左上角，中间和右上角三个数，其它数均可推导出来，更进一步，中间数为5，只需穷举两个数即可

#include <bits/stdc++.h>  
using namespace std;  
  
int main()  
{  
  int a,b,c=0;  
  for(a=1; a<9; a++)  
    for(b=1; b<9; b++)  
      if(a!=b && a!=5 && b!=5 && 5+a-b<=9 && 5+a-b>=1 && a+b-5<=9 && a+b-5>=1 && a*(15-a-b)*b*(5+b-a)*5*(5+a-b)*(10-b)*(a+b-5)*(10-a)==362880)  
        c+=1;  
  cout<<c<<"\n";  
  return 0;  
}
